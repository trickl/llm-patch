import java.util.*;

public class ExpressionEvaluator {
    public static void main(String[] args) {
        System.out.println(evaluate("3 + 4 * (2 - 1)")); // Should print 7
        System.out.println(evaluate("2 * 3 + 4")); // Should print 10
        System.out.println(evaluate("2 * (3 + 4)")); // Should print 14
        System.out.println(evaluate("8 / 2 * (2 + 2)")); // Should print 16
    }

    public static int evaluate(String expression) {
        return new Parser(new Tokenizer(expression)).parse();
    }

    private static class Tokenizer {
        private final String input;
        private int pos = 0;

        Tokenizer(String input) {
            this.input = input;
        }

        List<Object> tokenize() {
            List<Object> tokens = new ArrayList<>();
            while (pos < input.length()) {
                char ch = input.charAt(pos);
                if (Character.isDigit(ch)) {
                    tokens.add(parseNumber());
                } else if (ch == '+' || ch == '-' || ch == '*' || ch == '/') {
                    tokens.add(ch);
                    pos++;
                } else if (ch == '(') {
                    tokens.add('(');
                    pos++;
                } else if (ch == ')') {
                    tokens.add(')');
                    pos++;
                } else if (Character.isWhitespace(ch)) {
                    pos++;
                } else {
                    throw new IllegalArgumentException("Unexpected character: " + ch);
                }
            }
            return tokens;
        }

        private int parseNumber() {
            int start = pos;
            while (pos < input.length() && Character.isDigit(input.charAt(pos))) {
                pos++;
            }
            return Integer.parseInt(input.substring(start, pos));
        }
    }

    private static class Parser {
        private final List<Object> tokens;
        private int pos = 0;

        Parser(List<Object> tokens) {
            this.tokens = tokens;
        }

        int parse() {
            return parseExpression();
        }

        private int parseExpression() {
            int result = parseTerm();
            while (pos < tokens.size()) {
                char op = (char) tokens.get(pos);
                if (op == '+' || op == '-') {
                    pos++;
                    int nextTerm = parseTerm();
                    if (op == '+') {
                        result += nextTerm;
                    } else {
                        result -= nextTerm;
                    }
                } else {
                    break;
                }
            }
            return result;
        }

        private int parseTerm() {
            int result = parseFactor();
            while (pos < tokens.size()) {
                char op = (char) tokens.get(pos);
                if (op == '*' || op == '/') {
                    pos++;
                    int nextFactor = parseFactor();
                    if (op == '*') {
                        result *= nextFactor;
                    } else {
                        result /= nextFactor;
                    }
                } else {
                    break;
                }
            }
            return result;
        }

        private int parseFactor() {
            Object token = tokens.get(pos);
            pos++;
            if (token instanceof Integer) {
                return (Integer) token;
            } else if (token == '(') {
                int result = parseExpression();
                if (pos < tokens.size() && tokens.get(pos) == ')') {
                    pos++;
                } else {
                    throw new IllegalArgumentException("Expected )");
                }
                return result;
            } else if (token == '-') {
                return -parseFactor();
            } else {
                throw new IllegalArgumentException("Unexpected token: " + token);
            }
        }
    }
}
