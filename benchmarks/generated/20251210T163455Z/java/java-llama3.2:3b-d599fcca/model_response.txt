import java.util.*;

public class ExpressionEvaluator {

    public static void main(String[] args) {
        System.out.println(evaluate("1 + 2")); // 3
        System.out.println(evaluate("2 * 3 + 4")); // 10
        System.out.println(evaluate("2 * (3 + 4)")); // 14
        System.out.println(evaluate("8 / 2 * (2 + 2)")); // 16
    }

    public static int evaluate(String expression) {
        return new ExpressionEvaluator().evaluate(expression);
    }

    private static class Token {
        String type;
        Object value;

        public Token(String type, Object value) {
            this.type = type;
            this.value = value;
        }
    }

    private static class Parser {
        List<Token> tokens;
        int pos;

        public Parser(List<Token> tokens) {
            this.tokens = tokens;
            this.pos = 0;
        }

        Token parse() {
            if (pos >= tokens.size()) throw new RuntimeException("Unexpected end of input");
            return tokens.get(pos++);
        }
    }

    private static class Evaluator {
        List<Token> tokens;

        public Evaluator(List<Token> tokens) {
            this.tokens = tokens;
        }

        int evaluate() {
            while (true) {
                Token token = parse();
                if (token.type.equals("number")) {
                    return (int) token.value;
                } else if (token.type.equals("+") || token.type.equals("-")) {
                    int left = evaluate();
                    int right = evaluate();
                    if (token.type.equals("+")) return left + right;
                    return left - right;
                } else if (token.type.equals("*") || token.type.equals("/")) {
                    int left = evaluate();
                    int right = evaluate();
                    if (token.type.equals("*")) return left * right;
                    return left / right;
                } else if (token.type.equals("(")) {
                    Token rightParen = parse();
                    if (!rightParen.type.equals(")")) throw new RuntimeException("Unbalanced parentheses");
                    return evaluate();
                }
            }
        }

        private Token parse() {
            if (pos >= tokens.size()) throw new RuntimeException("Unexpected end of input");
            return tokens.get(pos++);
        }
    }

    public int evaluate(String expression) {
        List<Token> tokens = tokenize(expression);
        Evaluator evaluator = new Evaluator(tokens);
        return evaluator.evaluate();
    }

    private static List<Token> tokenize(String expression) {
        List<Token> tokens = new ArrayList<>();
        StringBuilder currentToken = new StringBuilder();

        for (char c : expression.toCharArray()) {
            if (Character.isDigit(c)) {
                currentToken.append(c);
            } else if ("+-*/()".indexOf(c) != -1) {
                if (currentToken.length() > 0) {
                    tokens.add(new Token(currentToken.toString(), Integer.parseInt(currentToken.toString())));
                    currentToken.setLength(0);
                }
                tokens.add(new Token(String.valueOf(c), c));
            }
        }

        if (currentToken.length() > 0) {
            tokens.add(new Token(currentToken.toString(), Integer.parseInt(currentToken.toString())));
        }

        return tokens;
    }
}
