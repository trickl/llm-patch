import java.util.*;

public class ExpressionEvaluator {

    public static void main(String[] args) {
        System.out.println(evaluate("1 + 2")); // 3
        System.out.println(evaluate("2 * 3 + 4")); // 10
        System.out.println(evaluate("2 * (3 + 4)")); // 14
        System.out.println(evaluate("8 / 2 * (2 + 2)")); // 16
    }

    public static int evaluate(String expression) {
        return new ExpressionEvaluator().evaluate(expression);
    }

    private static class Tokenizer implements Iterable<String> {

        private final String expression;
        private int pos = 0;

        public Tokenizer(String expression) {
            this.expression = expression;
        }

        @Override
        public Iterator<String> iterator() {
            while (pos < expression.length()) {
                if (Character.isDigit(expression.charAt(pos))) {
                    buildToken();
                }
                pos++;
            }
            return Collections.emptyIterator();
        }

        private void buildToken() {
            int end = pos;
            while (end < expression.length() && Character.isDigit(expression.charAt(end))) {
                end++;
            }
            yield(expression.substring(pos, end));
            pos = end + 1;
        }

        @Override
        public boolean hasNext() {
            return pos < expression.length();
        }
    }

    private static class Parser {

        private final Tokenizer tokenizer;

        public Parser(Tokenizer tokenizer) {
            this.tokenizer = tokenizer;
        }

        public int parse() throws Exception {
            return parseExpression();
        }

        private int parseExpression() throws Exception {
            int left = parseTerm();
            while (tokenizer.hasNext()) {
                String token = tokenizer.next();
                if (token.equals("+") || token.equals("-")) {
                    char op = token.charAt(0);
                    int right = parseTerm();
                    if (op == '+') {
                        return left + right;
                    } else {
                        return left - right;
                    }
                }
            }
            return left;
        }

        private int parseTerm() throws Exception {
            int left = parseFactor();
            while (tokenizer.hasNext()) {
                String token = tokenizer.next();
                if (token.equals("*") || token.equals("/")) {
                    char op = token.charAt(0);
                    int right = parseFactor();
                    if (op == '*') {
                        return left * right;
                    } else {
                        return left / right;
                    }
                }
            }
            return left;
        }

        private int parseFactor() throws Exception {
            if (tokenizer.hasNext()) {
                String token = tokenizer.next();
                if (token.equals("-")) {
                    return -parseFactor();
                }
            }
            return Integer.parseInt(tokenizer.next());
        }
    }

    public static class ExpressionEvaluator {

        public int evaluate(String expression) {
            try {
                Tokenizer tokenizer = new Tokenizer(expression);
                Parser parser = new Parser(tokenizer);
                return parser.parse();
            } catch (Exception e) {
                throw new RuntimeException("Invalid expression", e);
            }
        }
    }
}
