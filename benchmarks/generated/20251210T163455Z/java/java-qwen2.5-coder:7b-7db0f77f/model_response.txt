import java.util.*;

public class ExpressionEvaluator {
    public static void main(String[] args) {
        System.out.println(evaluate("3 + 4 * (2 - 1)")); // Output: 7
        System.out.println(evaluate("2 * 3 + 4")); // Output: 10
        System.out.println(evaluate("2 * (3 + 4)")); // Output: 14
        System.out.println(evaluate("8 / 2 * (2 + 2)")); // Output: 16
    }

    public static int evaluate(String expression) {
        return new Parser(new Tokenizer(expression)).parse();
    }
}

class Tokenizer {
    private final String input;
    private int pos = 0;

    public Tokenizer(String input) {
        this.input = input;
    }

    public List<Object> tokenize() {
        List<Object> tokens = new ArrayList<>();
        while (pos < input.length()) {
            char ch = input.charAt(pos);
            if (Character.isDigit(ch)) {
                int num = 0;
                boolean isNegative = false;
                if (ch == '-') {
                    isNegative = true;
                    pos++;
                    ch = input.charAt(pos);
                }
                while (pos < input.length() && Character.isDigit(input.charAt(pos))) {
                    num = num * 10 + (input.charAt(pos) - '0');
                    pos++;
                }
                tokens.add(isNegative ? -num : num);
            } else if (ch == '+' || ch == '-' || ch == '*' || ch == '/' || ch == '(' || ch == ')') {
                tokens.add(ch);
                pos++;
            } else {
                pos++;
            }
        }
        return tokens;
    }
}

class Parser {
    private final List<Object> tokens;
    private int pos = 0;

    public Parser(List<Object> tokens) {
        this.tokens = tokens;
    }

    public int parse() {
        return expr();
    }

    private int expr() {
        int result = term();
        while (pos < tokens.size()) {
            Object token = tokens.get(pos);
            if (token == '+') {
                pos++;
                result += term();
            } else if (token == '-') {
                pos++;
                result -= term();
            } else {
                break;
            }
        }
        return result;
    }

    private int term() {
        int result = factor();
        while (pos < tokens.size()) {
            Object token = tokens.get(pos);
            if (token == '*') {
                pos++;
                result *= factor();
            } else if (token == '/') {
                pos++;
                result /= factor();
            } else {
                break;
            }
        }
        return result;
    }

    private int factor() {
        if (pos < tokens.size()) {
            Object token = tokens.get(pos);
            if (token instanceof Integer) {
                pos++;
                return (Integer) token;
            } else if (token == '(') {
                pos++;
                int result = expr();
                if (pos < tokens.size() && tokens.get(pos) == ')') {
                    pos++;
                    return result;
                }
            } else if (token == '-') {
                pos++;
                return -factor();
            }
        }
        throw new IllegalArgumentException("Invalid expression");
    }
}
