import re

class Token:
    def __init__(self, type, value):
        self.type = type
        self.value = value

class Tokenizer:
    def tokenize(self, expression):
        tokens = []
        current_token = None
        
        for char in expression:
            if char.isspace():
                continue
            
            if not current_token or (current_token.type == 'NUMBER' and char.isdigit()):
                current_token = Token('NUMBER', char)
            elif not current_token or (current_token.type != 'OPERATOR' and char in '+-*/()'):
                tokens.append(current_token)
                current_token = Token(char, char)
            else:
                if current_token.value + char in '+-*/':
                    tokens.append(current_token)
                    current_token = Token('OPERATOR', char)
        
        if current_token:
            tokens.append(current_token)
        
        return tokens

class Parser:
    def __init__(self):
        self precedence = {'+': 1, '-': 1, '*': 2, '/': 2}
    
    def parse(self, tokens):
        output_queue = []
        operator_stack = []
        
        for token in tokens:
            if token.type == 'NUMBER':
                output_queue.append(token.value)
            elif token.type == 'OPERATOR':
                while (operator_stack and 
                       self.precedence[operator_stack[-1]] >= self.precedence[token.value]):
                    output_queue.append(operator_stack.pop())
                operator_stack.append(token.value)
        
        while operator_stack:
            output_queue.append(operator_stack.pop())
        
        return output_queue

class Evaluator:
    def __init__(self):
        pass
    
    def evaluate(self, expression):
        stack = []
        current_token = None
        
        for char in expression:
            if char.isspace():
                continue
            
            if not current_token or (current_token.type == 'NUMBER' and char.isdigit()):
                current_token = Token('NUMBER', char)
            elif not current_token or (current_token.type != 'OPERATOR' and char in '+-*/()'):
                tokens = self.tokenize(char)
                stack.append(self.parse(tokens))
                current_token = None
            else:
                if current_token.value + char in '+-*/':
                    tokens = self.tokenize(char)
                    stack.append(self.parse(tokens))
                    current_token = None
        
        result = stack.pop()
        
        if isinstance(result, str):
            return int(result)
        elif isinstance(result, Token) and result.type == 'NUMBER':
            return int(result.value)
        else:
            raise ValueError("Invalid expression")

def main():
    tokenizer = Tokenizer()

    evaluator = Evaluator()
    
    expressions = ["1 + 2", "2 * 3 + 4", "2 * (3 + 4)", "8 / 2 * (2 + 2)"]
    
    for expression in expressions:
        print(f"Expression: {expression}")
        try:
            result = evaluator.evaluate(expression)
            print(f"Result: {result}")
        except ValueError as e:
            print(f"Error: {e}")

if __name__ == '__main__':
    main()
